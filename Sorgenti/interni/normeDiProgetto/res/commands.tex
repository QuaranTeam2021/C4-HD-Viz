{
	\begin{center}
		\begin{LongTable}{#1}{#2}{x{.14\hsize}|x{.12\hsize}|p{.26\hsize}|x{.38\hsize}}
			\topline
			\rowcolor{headercolour}
			\lhdr{Codice} & \lhdr{Nome} &  \lhdr{Descrizione} & \lhdr{Formula} \\ \capsep
			\endfirsthead
			\continuedcaption
			\topline
			\rowcolor{headercolour}
			\lhdr{Codice} & \lhdr{Nome} & \lhdr{Descrizione} & \lhdr{Formula} \\ \capsep
			\endhead
			\rowsep
			\defaultfooter{4}
		}
		{
			\bottomline
		\end{LongTable}
	\end{center}
}
